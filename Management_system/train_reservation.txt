class Train:
    def __init__(self,train_name,source,destination,total_seats,available_seats):
        self.train_name = train_name
        self.source = source
        self.destination = destination 
        self.total_seats = total_seats
        self.available_seats = available_seats

    def __str__(self):
        return f"|TRAIN_NAME: {self.train_name}| Source: {self.source}| Destination: {self.destination}| Total_seats: {self.total_seats}| Available_seasts: {self.available_seats}|"

        print("\n1.Export CSV \n2.Import CSV")
        ch = int(input("Enter your choice:"))

        if ch == 1:
            try:

                cursor.execute("select * from trains")
                rows = cursor.fetchall()

                with open("trains.csv","w",newline="") as f:
                    writer = csv.writer(f)
                    writer.writerow(["train_id","train_name","source","destination","total_seats","available_seats"])
                    writer.writerows(rows)
                print("Data has been exported to csv format!")
            
            except Exception as e:
                print("Error:",e)

        elif ch == 2:

            try:

                with open("trains.csv","r") as f:
                    reader = csv.reader(f)
                    next(reader)

                    for row in reader:
                        cursor.execute("insert into trains(train_name, source, destination, total_seats, available_seats) values(?,?,?,?)",(row[1],row[2],row[3],row[4],row[5]))
                    
                    print("Data has been imported successfully!")
            except Exception as e:
                print("Error:",e)
        else:
            print("Invalid choice!")
